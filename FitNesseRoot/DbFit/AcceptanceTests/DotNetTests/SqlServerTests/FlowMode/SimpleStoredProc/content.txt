This stored proc tests output parameters - CalcLength returns string length as second output param
{{{
CREATE PROCEDURE CalcLength_P
@name VARCHAR(255)
, @strlength INT OUTPUT
AS
BEGIN
	SET @strlength = DataLength(@name);
END;
GO
}}}
